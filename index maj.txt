function ci(n){let h=0;for(let i=0;i<n.length;i++)h=(h*31+n.charCodeAt(i))%CBG.length;return h;}

async function api(action, data={}) {
  try {
    const res = await fetch(API_URL, {
      method: "POST",
      body: JSON.stringify({action, ...data})
    });
    const json = await res.json();
    if (json.error) throw new Error(json.error);
    return json;
  } catch(e) {
    showError("Erreur de connexion : " + e.message + ". Vérifiez que l'URL Apps Script est correcte.");
    throw e;
  }
}

async function loadAll() {
  try {
    const data = await api("getAll");
    events = data.events || [];
    conseillers = data.conseillers || [];
    updateTopbar(); updateStats(); render();
  } catch(e) {}

function showToast(msg) {

  t.textContent = msg; t.classList.add("show");

function setTab(t) {
  currentTab = t; activeFilter = "all";
  ["upcoming","past","annual","manage"].forEach(id => {
    document.getElementById("tab-"+id).style.display = id===t ? "block" : "none";
  });
  document.querySelectorAll(".tab").forEach((b,i) => b.classList.toggle("active", ["upcoming","past","annual","manage"][i]===t));
  if(t==="manage") renderManage();
  else if(t==="annual") renderAnnual();
  else render();
}

function updateTopbar() {
  const av = document.getElementById("top-av"), lbl = document.getElementById("top-lbl");
  if(currentUser) {

    av.style.background = CBG[idx]; av.style.color = CTX[idx];

    av.innerHTML = '<i class="ti ti-user" style="font-size:13px"></i>';
    av.style.background=""; av.style.color=""; av.classList.add("anon");
    lbl.textContent = "Choisir mon nom"; lbl.style.color = "";
  }

function updateStats() {
  const up = events.filter(e => { const d=new Date(e.date); d.setHours(0,0,0,0); return d>=TODAY; });
  document.getElementById("stat-ev").textContent = up.length;
  document.getElementById("stat-pr").textContent = events.filter(e => currentUser && e.participants.includes(currentUser)).length;

function renderFilterBar(list, barId) {
  const types = [...new Set(list.map(e => e.type||"autre"))];
  const bar = document.getElementById(barId);
  if(types.length < 2) { bar.innerHTML=""; return; }
  bar.innerHTML = '<button class="filter-btn'+(activeFilter==="all"?" active":"")+'" onclick="setFilter(\'all\',\''+barId+'\')">Tous</button>'
    + types.map(t => {
        const tp = TYPES[t]||TYPES.autre;
        const isActive = activeFilter===t;
        return `<button class="filter-btn${isActive?" active":""}" onclick="setFilter('${t}','${barId}')" style="${isActive?`background:${tp.bg};border-color:${tp.border};color:${tp.color}`:""}">${tp.emoji} ${tp.label}</button>`;
      }).join("");
}

function setFilter(f, barId) {
  activeFilter = f;
  const list = currentTab==="upcoming"
    ? events.filter(e => { const d=new Date(e.date); d.setHours(0,0,0,0); return d>=TODAY; })
    : events.filter(e => { const d=new Date(e.date); d.setHours(0,0,0,0); return d<TODAY; });
  renderFilterBar(list, barId);
  renderEvents(list, currentTab==="upcoming"?"events-upcoming":"events-past");
}

function renderEvents(list, cid) {
  const el = document.getElementById(cid);
  let filtered = activeFilter==="all" ? list : list.filter(e => (e.type||"autre")===activeFilter);
  filtered = [...filtered].sort((a,b) => a.date.localeCompare(b.date));
  if(!filtered.length) { el.innerHTML='<div class="empty-state"><i class="ti ti-calendar-off"></i><p>Aucun événement ici.</p></div>'; return; }
  el.innerHTML = filtered.map(ev => {
    const d = new Date(ev.date), isPast=d<TODAY, isMe=currentUser&&ev.participants.includes(currentUser), isOpen=openCards.has(ev.id);
    const tp = TYPES[ev.type||"autre"]||TYPES.autre;
    const chips = ev.participants.length
      ? ev.participants.map(p => { const idx=ci(p); return `<span class="chip${p===currentUser?" me":""}"><span class="chip-av" style="${p!==currentUser?`background:${CBG[idx]};color:${CTX[idx]}`:""}">${ini(p)}</span>${p}</span>`; }).join("")
      : '<span style="font-size:12px;color:var(--text3)">Aucun participant encore</span>';
    return `<div class="ev-card${isMe?" is-present":""}${isPast?" is-past":""}" id="card-${ev.id}">
      <div class="ev-top" onclick="toggleCard('${ev.id}')">
        <div class="ev-stripe" style="background:${tp.color};opacity:0.7"></div>
        <div class="ev-date-badge"><div class="ev-day">${d.getDate()}</div><div class="ev-month">${MS[d.getMonth()]}</div></div>
        <div class="ev-info">
          <div class="ev-name">${ev.name}</div>
          <div class="ev-meta">

          ${!isPast&&currentUser?`<button class="btn-join ${isMe?"in":"out"}" onclick="event.stopPropagation();togglePresence('${ev.id}')">${isMe?'<i class="ti ti-check" style="font-size:12px;margin-right:3px"></i>Présent·e':'+ M\'inscrire'}</button>`:""}

function render() {
  const up = events.filter(e => { const d=new Date(e.date); d.setHours(0,0,0,0); return d>=TODAY; });
  const past = events.filter(e => { const d=new Date(e.date); d.setHours(0,0,0,0); return d<TODAY; });
  if(currentTab==="upcoming") { renderFilterBar(up,"filter-bar-up"); renderEvents(up,"events-upcoming"); }
  if(currentTab==="past") { renderFilterBar(past,"filter-bar-past"); renderEvents(past,"events-past"); }
  updateStats();
}

function renderAnnual() {
  document.getElementById("year-lbl").textContent = currentYear;
  const wrap = document.getElementById("annual-wrap");
  const yev = events.filter(e => new Date(e.date).getFullYear()===currentYear);
  if(!yev.length) { wrap.innerHTML='<div class="empty-state"><i class="ti ti-calendar-off"></i><p>Aucun événement en '+currentYear+'.</p></div>'; return; }
  const byM = {};
  yev.forEach(ev => { const m=new Date(ev.date).getMonth(); if(!byM[m])byM[m]=[]; byM[m].push(ev); });
  wrap.innerHTML = Object.keys(byM).map(Number).sort((a,b)=>a-b).map(m => {
    const rows = [...byM[m]].sort((a,b)=>a.date.localeCompare(b.date)).map(ev => {
      const d=new Date(ev.date), hasP=ev.participants.length>0, cl=hasP?"green":"red", tp=TYPES[ev.type||"autre"]||TYPES.autre;
      return `<div class="annual-row ${cl}" onclick="goToEvent('${ev.id}')">
        <span class="a-dot ${cl}"></span>
        <span class="a-day ${cl}">${String(d.getDate()).padStart(2,"0")}</span>
        <span class="a-name ${cl}">${ev.name}</span>
        <span class="a-type" style="background:${tp.bg};color:${tp.color}">${tp.emoji} ${tp.label}</span>
        <span class="a-badge ${cl}">${hasP?ev.participants.length+" présent"+(ev.participants.length>1?"s":""):"Personne"}</span>
      </div>`;
    }).join("");
    return `<div><div class="month-hd">${ML[m]}</div>${rows}</div>`;
  }).join("");
}

function changeYear(d) { currentYear+=d; renderAnnual(); }

function goToEvent(id) {
  setTab("upcoming"); activeFilter="all"; openCards.add(id); render();
  setTimeout(() => { const el=document.getElementById("card-"+id); if(el)el.scrollIntoView({behavior:"smooth",block:"center"}); }, 120);

function toggleCard(id) {
  if(openCards.has(id))openCards.delete(id); else openCards.add(id);
  const b=document.getElementById("body-"+id); if(b)b.classList.toggle("open",openCards.has(id));
}

async function togglePresence(id) {
  if(!currentUser) { openIdentityModal(); return; }
  const ev = events.find(e=>e.id===id); if(!ev) return;
  try {
    const res = await api("togglePresence", {id, user: currentUser});
    ev.participants = res.participants;
    render(); if(currentTab==="annual")renderAnnual();
    showToast(ev.participants.includes(currentUser) ? "✓ Présence confirmée !" : "Présence retirée");
  } catch(e) {}
}

function confirmDeleteEvent(id, name) {
  document.getElementById("confirm-title").textContent = "Supprimer l'événement";
  document.getElementById("confirm-msg").textContent = `Supprimer « ${name} » ? Cette action est irréversible.`;
  document.getElementById("confirm-ok").onclick = async () => {
    try {
      await api("deleteEvent", {id});
      events = events.filter(e=>e.id!==id); openCards.delete(id);
      closeModal("confirm-modal"); render(); if(currentTab==="annual")renderAnnual();
      showToast("Événement supprimé");
    } catch(e) {}
  };
  document.getElementById("confirm-modal").classList.add("open");
}

function selectUser(name) {
  currentUser=name; localStorage.setItem("shh_user",name);

function openAddEvent() {
  ["ev-name-in","ev-lieu-in"].forEach(id=>document.getElementById(id).value="");
  document.getElementById("ev-date-in").value="";
  document.getElementById("ev-type-in").value="conseil";
  document.getElementById("add-event-modal").classList.add("open");
  setTimeout(()=>document.getElementById("ev-name-in").focus(),100);
}

async function saveEvent() {
  const name=document.getElementById("ev-name-in").value.trim();
  const date=document.getElementById("ev-date-in").value;
  const type=document.getElementById("ev-type-in").value;
  const lieu=document.getElementById("ev-lieu-in").value.trim();
  if(!name||!date) return;
  const btn=document.getElementById("save-ev-btn");
  btn.disabled=true; btn.textContent="Enregistrement…";
  try {
    const res = await api("addEvent",{name,type,date,lieu});
    events.push({id:res.id,name,type,date,lieu,participants:[]});
    closeModal("add-event-modal"); activeFilter="all";
    const d=new Date(date); d.setHours(0,0,0,0);
    if(d>=TODAY) setTab("upcoming"); else if(currentTab==="annual") renderAnnual(); else render();
    showToast("Événement ajouté !");
  } catch(e) {}
  btn.disabled=false; btn.innerHTML='<i class="ti ti-check" style="font-size:14px"></i>Enregistrer';
}

function closeModal(id) { document.getElementById(id).classList.remove("open"); }

async function addConseiller() {
  const inp=document.getElementById("new-c-input"), name=inp.value.trim();
  if(!name||conseillers.includes(name)) return;
  try {
    await api("addConseiller",{name});
    conseillers.push(name); inp.value="";
    updateStats(); renderManage();
    showToast("Conseiller ajouté !");
  } catch(e) {}
}

function confirmRemoveConseiller(name) {
  document.getElementById("confirm-title").textContent = "Retirer le conseiller";
  document.getElementById("confirm-msg").textContent = `Retirer ${name} de la liste ? Ses inscriptions aux événements seront supprimées.`;
  document.getElementById("confirm-ok").onclick = async () => {
    try {
      await api("deleteConseiller",{name});
      conseillers=conseillers.filter(c=>c!==name);
      events.forEach(ev=>{ev.participants=ev.participants.filter(p=>p!==name);});
      if(currentUser===name){currentUser="";localStorage.removeItem("shh_user");}
      closeModal("confirm-modal"); updateTopbar(); updateStats(); renderManage(); render();
      showToast("Conseiller retiré");
    } catch(e) {}
  };
  document.getElementById("confirm-modal").classList.add("open");
}

function renderManage() {
  document.getElementById("manage-count").textContent = `${conseillers.length} conseiller${conseillers.length>1?"s":""}`;
  document.getElementById("manage-list").innerHTML = conseillers.length
    ? conseillers.map(c => {
        const idx=ci(c), pres=events.filter(e=>e.participants.includes(c)).length;
        return `<div class="c-card">
          <div style="width:40px;height:40px;border-radius:50%;background:${CBG[idx]};color:${CTX[idx]};display:flex;align-items:center;justify-content:center;font-size:13px;font-weight:700;flex-shrink:0">${ini(c)}</div>
          <div class="c-info"><div class="c-name">${c}</div><div class="c-sub">${pres} présence${pres>1?"s":""} confirmée${pres>1?"s":""}</div></div>
          <button class="btn-trash" onclick="confirmRemoveConseiller('${c.replace(/'/g,"\\'")}')"><i class="ti ti-trash" style="font-size:14px"></i> Supprimer</button>
        </div>`;
      }).join("")
    : '<div class="empty-state"><i class="ti ti-users"></i><p>Aucun conseiller.</p></div>';
}
